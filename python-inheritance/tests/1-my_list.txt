def test_mylist():
    # Test 1 : liste vide
    l1 = MyList()
    print("Test 1 - Empty list:")
    print(l1)          # []
    l1.print_sorted()  # []

    # Test 2 : liste avec un seul élément
    l2 = MyList([10])
    print("Test 2 - Single element:")
    print(l2)          # [10]
    l2.print_sorted()  # [10]

    # Test 3 : liste déjà triée
    l3 = MyList([1, 2, 3, 4, 5])
    print("Test 3 - Already sorted:")
    print(l3)          # [1, 2, 3, 4, 5]
    l3.print_sorted()  # [1, 2, 3, 4, 5]

    # Test 4 : liste triée à l'envers
    l4 = MyList([5, 4, 3, 2, 1])
    print("Test 4 - Reverse sorted:")
    print(l4)          # [5, 4, 3, 2, 1]
    l4.print_sorted()  # [1, 2, 3, 4, 5]

    # Test 5 : liste avec éléments en désordre
    l5 = MyList([3, 1, 4, 1, 5, 9, 2])
    print("Test 5 - Random order:")
    print(l5)          # [3, 1, 4, 1, 5, 9, 2]
    l5.print_sorted()  # [1, 1, 2, 3, 4, 5, 9]

    # Test 6 : vérification que la liste d'origine ne change pas après print_sorted
    l6 = MyList([7, 3, 5])
    print("Test 6 - Original list unchanged:")
    print(l6)          # [7, 3, 5]
    l6.print_sorted()  # [3, 5, 7]
    print(l6)          # [7, 3, 5]

test_mylist()
